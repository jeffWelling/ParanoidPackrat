                def setDuplication name, guaranteed_num_of_backups
                        unless @Configs.include?(name)
                                puts "Must first addname('#{name}')"
                                PPConfig.checkYourConfig
                        end
                        @Configs[name][:GuaranteedNumBackups]==guaranteed_num_of_backups.to_i
                        true
                end
