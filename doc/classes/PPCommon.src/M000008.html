        def self.symbolize text
         return :nil if text.nil?
                return :empty if text.empty?
                return :quit if text =~ /^(q|quit)$/i
                return :edit if text =~ /^(e|edit)$/i
                return :yes  if text =~ /^(y|yes)$/i
                return :no   if text =~ /^(n|no)$/i
                text.to_sym
        end
