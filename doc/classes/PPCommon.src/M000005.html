        def self.makeBackupDirectory(dir)
                raise "You idiot" unless dir.class==String
                dir=PPCommon.addSlash(dir)
                #Make sure the directory is empty
                counter=0
                Find.find(dir) {|file|
                        counter+=1
                }
                return FALSE unless counter==1
                FileUtils.mkdir( dir + "backup/", 700 )[0]
        end
